    rho = sum(fIn);
    ux  = reshape ( (cx * reshape(fIn,9,lx*ly)), 1,lx,ly) ./rho;
    uy  = reshape ( (cy * reshape(fIn,9,lx*ly)), 1,lx,ly) ./rho;

    % COLLISION STEP
    for i=1:9
    	cu = 3*(cx(i)*ux+cy(i)*uy);
    	fEq(i,:,:)  = rho .* t(i) .* ...
                    ( 1 + cu + 1/2*(cu.*cu)  - 3/2*(ux.^2+uy.^2) );
       fOut(i,:,:) = fIn(i,:,:) - omega .* (fIn(i,:,:)-fEq(i,:,:));
    end

	% STREAMING STEP
